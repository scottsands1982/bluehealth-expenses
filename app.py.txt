# Hilfsfunktionen für die kostenlose Text-Extraktion (OCR)
def extract_data_from_text(text):
    text_lower = text.lower()
    
    # 1. Datum suchen (Formate: DD.MM.YYYY oder YYYY-MM-DD)
    date_match = re.search(r'(\b\d{2}\.\d{2}\.\d{4}\b)|(\b\d{4}-\d{2}-\d{2}\b)', text)
    detected_date = datetime.today()
    if date_match:
        try:
            detected_date = datetime.strptime(date_match.group(0).replace('-', '.'), '%d.%m.%Y')
        except:
            pass

    # 2. Endsumme suchen
    detected_amount = 0.0
    # Suche nach Zeilen mit typischen Keywords
    keywords = ["endsumme", "gesamtbetrag", "total", "summe", "zahlbetrag", "eur"]
    lines = text_lower.split('\n')
    for line in lines:
        if any(kw in line for kw in keywords):
            amounts = re.findall(r'\d+[\.,]\d{2}', line)
            if amounts:
                detected_amount = float(amounts[-1].replace(',', '.'))
                break
    
    # Fallback: Höchste Zahl finden, falls kein Keyword matcht
    if detected_amount == 0.0:
        all_amounts = re.findall(r'\d+[\.,]\d{2}', text_lower)
        if all_amounts:
            detected_amount = max([float(a.replace(',', '.')) for a in all_amounts])

    # 3. Kategorie erraten
    category = "Sonstiges"
    if any(k in text_lower for k in ["aral", "shell", "total", "jet", "tanken", "kraftstoff", "diesel", "super"]):
        category = "Tanken/Kraftstoff"
    elif any(k in text_lower for k in ["restaurant", "bewirtung", "cafe", "bistro", "essen"]):
        category = "Kunden-Verpflegung"
    elif any(k in text_lower for k in ["db", "bahn", "taxi", "hotel", "flug"]):
        category = "Reisekosten"
    elif any(k in text_lower for k in ["amazon", "staples", "papier", "büro"]):
        category = "Büromaterial"

    return detected_date, detected_amount, category

# Upload-Feld (Nutzt am iPhone automatisch die Kamera!)
uploaded_file = st.file_uploader("Beleg auswählen oder fotografieren...", type=["png", "jpg", "jpeg"])

if uploaded_file is not None:
    image = Image.open(uploaded_file)
    st.image(image, caption="Hochgeladener Beleg", width=300)
    
    with st.spinner("Lese Belegdaten aus..."):
        try:
            # Echtes OCR auf dem Server
            raw_text = pytesseract.image_to_string(image, lang="deu+eng")
            extracted_date, extracted_amount, guessed_category = extract_data_from_text(raw_text)
        except Exception as e:
            # Lokaler Fallback, falls Tesseract auf deinem PC noch nicht installiert ist
            st.warning("Lokaler Testmodus: OCR läuft vollautomatisch, sobald die App online ist!")
            extracted_date = datetime.today()
            extracted_amount = 85.40
            guessed_category = "Tanken/Kraftstoff"

    st.markdown("<h4 style='color: #2d2d2d;'>Gefundene Daten (Bitte überprüfen):</h4>", unsafe_allow_html=True)
    
    # Formular für manuelle Korrekturen (Dropdowns & Inputs)
    final_date = st.date_input("Datum des Belegs", extracted_date)
    final_amount = st.number_input("Endsumme in €", value=extracted_amount, step=0.01, format="%.2f")
    
    categories = ["Tanken/Kraftstoff", "Dienstwagen-Wartung", "Kunden-Verpflegung", "Büromaterial", "Reisekosten", "Sonstiges"]
    default_index = categories.index(guessed_category) if guessed_category in categories else 5
    
    final_category = st.selectbox("Kategorie (Dropdown zur Korrektur)", categories, index=default_index)

    # Absenden
    if st.button("Ausgabe für Buchhaltung freigeben", type="primary"):
        st.success(f"Erfolgreich erfasst: {final_amount} € für '{final_category}'!")
        
        # Kreatives Extra: Biete direkt einen Excel/CSV Download für den Mitarbeiter an
        data = [{"Datum": final_date, "Betrag_EUR": final_amount, "Kategorie": final_category}]
        df = pd.DataFrame(data)
        csv = df.to_csv(index=False).encode('utf-8')
        st.download_button("Excel/CSV-Export herunterladen", data=csv, file_name=f"Auslage_{final_date}.csv", mime="text/csv")
